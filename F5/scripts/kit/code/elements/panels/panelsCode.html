                // Delete panel
                     $scope.deletePanel = function (event) {
                        event.target.offsetParent.remove();
                    };
                // Expand panel 
                    $scope.expandPanel = function (event) {
                        var panel = $(event.target.parentNode.parentNode);
                        panel.toggleClass('expanded');
                    };
